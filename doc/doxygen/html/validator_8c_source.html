103 static bool validate_perm(char *filename, struct stat *filestat, parser_t *exp) {
104  (void)filename;
105  mode_t filemode = filestat->st_mode;
106  int refperm = *(int *)exp->value;
107  bool exact_perm = exp->comp == EXACT;
108  for (int i = 0; i < PERM_OPTIONS_COUNT; i++) {
109  bool req_perm = !!(refperm & PERM_OPTIONS[i]);
110  bool file_perm = !!(filemode & PERM_FLAGS[i]);
111  if (req_perm != file_perm && (exact_perm || !file_perm))
112  return false; // perm mismatch
113  }
114  return true;
115 }
