264 static parser_t *parse_token(char **expression[], size_t *size) {
265  char *exp = *expression[0];
266  *expression = *expression + 1;
267  (*size)--;
268  // TODO: handle parenthesis
269  for (int i = 0; i < OPERATORS_COUNT; i++) // an operator
270  if (strcmp(exp, operator[i]) == 0)
271  return parse_op(operator_type[i]);
272 
273  if (!*size) // end of expression
274  return NULL;
275 
276  char *exp_param = *expression[0];
277  *expression = *expression + 1;
278  (*size)--;
279  for (int i = 0; i < CRITERIA_COUNT; i++) // a criteria
280  if (strcmp(exp, criteria[i]) == 0)
281  return criteria_type[i](exp_param);
282 
283  return NULL; // invalid token
284 }

310 static parser_t *parser_infix_notation(parser_t *exp) {
311  parser_t *output = NULL;
312  parser_t *stack = NULL;
313 
314  while (exp) {
315  if (exp->crit & CRITERIA) // is a criteria
316  move_token(&output, &exp);
317  else if (exp->crit & OPERATOR) { // an operator
318  while (stack && stack->crit & OPERATOR && stack->crit > exp->crit) move_token(&output, &stack);
319  move_token(&stack, &exp);
320  } else if (exp->crit == LPARENTHESIS)
321  move_token(&stack, &exp);
322  else if (exp->crit == RPARENTHESIS) {
323  while (stack && stack->crit != LPARENTHESIS) move_token(&output, &stack);
324  if (!stack)
325  return NULL; // unmatched right parenthesis
326  stack = stack->next;
327  exp = exp->next;
328  }
329  }
330  if (stack) {
331  if (stack->crit & PARENTHESIS) // unmatched left parenthesis
332  return NULL;
333  while (stack != NULL) move_token(&output, &stack);
334  }
335 
336  return output;
337 }

339 parser_t *parser_parse(char *expression[], size_t size) {
340  parser_t *res;
341  parser_t *last;
342  res = last = NULL;
343  while (size) {
344  parser_t *parsed_token = parse_token(&expression, &size);
345  if (!parsed_token)
346  return NULL;
347 
348  // inject AND operator between consequitive criteria
349  if (last && (last->crit & CRITERIA || last->crit == RPARENTHESIS) &&
350  (parsed_token->crit & CRITERIA || parsed_token->crit == LPARENTHESIS || parsed_token->crit == NOT)) {
351  parser_t *and_token = parse_op(AND);
352  last->next = and_token;
353  last = and_token;
354  }
355  if (!res)
356  res = last = parsed_token;
357  else {
358  last->next = parsed_token;
359  last = parsed_token;
360  }
361  }
362 
363  return parser_infix_notation(res);
364 }
