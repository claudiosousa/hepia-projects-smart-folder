196 parser_t *parse_token(char **expression[], size_t *size) {
197  char *exp = *expression[0];
198  *expression = *expression + 1;
199  (*size)--;
200  // TODO: handle parenthesis
201  for (int i = 0; i < OPERATORS_COUNT; i++) // an operator
202  if (strcmp(exp, operator[i]) == 0)
203  return parse_op(operator_type[i]);
204 
205  if (!*size) // end of expression
206  return NULL;
207 
208  char *exp_param = *expression[0];
209  *expression = *expression + 1;
210  (*size)--;
211  for (int i = 0; i < CRITERIA_COUNT; i++) // a criteria
212  if (strcmp(exp, criteria[i]) == 0)
213  return criteria_type[i](exp_param);
214 
215  return NULL; // invalid token
216 }

226  Orders the parsed expression and per the Shunting yard algorithm
227  (https://en.wikipedia.org/wiki/Shunting-yard_algorithm)
228 */
229 parser_t *parser_infix_notation(parser_t *exp) {
230  parser_t *output = NULL;
231  parser_t *stack = NULL;
232 
233  while (exp) {
234  if (exp->crit & CRITERIA) // is a criteria
235  move_token(&output, &exp);
236  else if (exp->crit & OPERATOR) { // an operator
237  while (stack && stack->crit & OPERATOR && stack->crit > exp->crit) move_token(&output, &stack);
238  move_token(&stack, &exp);
239  } else if (exp->crit == LPARENTHESIS)
240  move_token(&stack, &exp);
241  else if (exp->crit == RPARENTHESIS) {
242  while (stack && stack->crit != LPARENTHESIS) move_token(&output, &stack);
243  if (!stack)
244  return NULL; // unmatched right parenthesis
245  stack = stack->next;
246  exp = exp->next;
247  }
248  }
249  if (stack) {
250  if (stack->crit & PARENTHESIS) // unmatched left parenthesis
251  return NULL;
252  while (stack != NULL) move_token(&output, &stack);
253  }
254 
255  return output;
256 }

258 parser_t *parser_parse(char *expression[], size_t size) {
259  parser_t *res;
260  parser_t *last;
261  res = last = NULL;
262  while (size) {
263  parser_t *parsed_token = parse_token(&expression, &size);
264  if (!parsed_token)
265  return NULL;
266 
267  // inject AND operator between consequitive criteria
268  if (last && (last->crit & CRITERIA || last->crit == RPARENTHESIS) &&
269  (parsed_token->crit & CRITERIA || parsed_token->crit == LPARENTHESIS || parsed_token->crit == NOT)) {
270  parser_t *and_token = parse_op(AND);
271  last->next = and_token;
272  last = and_token;
273  }
274  if (!res)
275  res = last = parsed_token;
276  else {
277  last->next = parsed_token;
278  last = parsed_token;
279  }
280  }
281 
282  return parser_infix_notation(res);
283 }
