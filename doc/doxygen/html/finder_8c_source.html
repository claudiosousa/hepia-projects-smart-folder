101  struct stat file_stat;
102 
103  switch (dent->d_type) {
104  case DT_DIR:
105  if (strcmp(dent->d_name, ".") != 0 && strcmp(dent->d_name, "..") != 0)
106  filelist = finder_find_in_dir(dirpath, filelist, expression);
107  break;
108  case DT_LNK:
109  stat(dirpath, &file_stat);

111  filelist = finder_find_in_dir(dirpath, filelist, expression);
112  else if (validator_validate(dent->d_name, &file_stat, expression)) {
113  if (!finder_hash_exist(file_stat.st_ino)) {
114  finder_hash_add(file_stat.st_ino);

116  }
117  }
118  break;
119  case DT_REG:

121  if (!finder_hash_exist(file_stat.st_ino) && validator_validate(dent->d_name, &file_stat, expression)) {
122  finder_hash_add(dent->d_ino);
123  filelist = finder_add_found_file(dirpath, filelist);
124  }
125  }
